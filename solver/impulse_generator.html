def ricker(dt, pt):
    nt = int(2 * pt / dt)
    c = np.zeros(nt)
    t0 = pt / dt
    a_ricker = 4 / pt

    for it in range(0, nt):
        t = ((it + 1) - t0) * dt
        c[it] = -2 * a_ricker * t * np.exp(-(a_ricker * t) ** 2)

    return c
dt = 1/44100
T0 = 1/200

tmp = np.diff(tmp)

def gaus2d(
    x: np.ndarray,
    y: np.ndarray,
    A: float = 1, # amplitude
    mx: float = 0, # center x
    my: float = 0, # center y
    sx: float = 1, # sigma x (width in x direction)
    sy: float = 1, # sigma y (width in y direction)
):

    # norm = 1.0 / (2.0 * np.pi * sx * sy)
    gaussian = np.exp(
        -((x - mx) ** 2.0 / (2.0 * sx**2.0) + (y - my) ** 2.0 / (2.0 * sy**2.0))
    )
    return A * gaussian

y = np.linspace(0, 50)
x = np.linspace(0, 50)
x, y = np.meshgrid(x, y) # get 2D variables instead of 1D
z = gaus2d(x, y, 2, 5, 5)

print(z.max())
